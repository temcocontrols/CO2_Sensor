1 function snrdB = fft_fp(X)
 2% newFFT: - returns the discrete Fourier transform (DFT) of vector X,
 3% computed with a fast Fourier transform (FFT) algorithm. If X is a matrix,
 4% newFFT returns the Fourier transform of each column of the matrix
 5
 6% Syntax: Y = newFFT(X,N)
 7%
 8% Inputs:
 9%    X          - Time domain input signal
10%    N          - Number of FFT point
11%
12% Outputs:
13%    Y      - Frequency domain N point DFT
14%
15% Other m-files required: 
16%   Subfunctions: 
17%   Upperfuncions: 
18%   MAT-files required: 
19%
20
21%------------- BEGIN CODE --------------
22
23
24N=64;
25[m n]= size(X);
26
27% If the length of X is greater than n, the sequence X is truncated.
28if m>N
29    Xmod= X(1:N,:);
30% If the length of X is less than N, X is padded with trailing zeros to
31% length N
32elseif m<N
33    Xmod =[X ;zeros(N-m,n)];
34    
35elseif m==N
36    Xmod=X;
37end
38
39% j   = sqrt(-1);
40j   = mpower(-1,0.5);
41tk  = 0:(N-1);
42fn  = transpose(0:(N-1))/N;
43wn  = 2*pi*fn;
44
45p= -j*wn*tk;
46
47Wp  = exp(p); %DFT Matrix
48Y   = Wp*Xmod;
49
50delta=160e-3;
51fs=1/delta;
52FFT_sampled_data=1/m*Y;
53absFFT_sampled_data=abs(FFT_sampled_data);
54signal_value=max(absFFT_sampled_data);
55noise_mean=(sum(absFFT_sampled_data)-signal_value)/(m-1);    
56snrVal=signal_value/noise_mean;
57snrdB=10*log10(snrVal); 
58
59%------------- END OF CODE --------------
